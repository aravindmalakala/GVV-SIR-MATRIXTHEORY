\textbf{Question}:\\
Write the direction ratios of the vector $\overrightarrow{a}=\hat{i}+\hat{j}-\hat{k}$ and hence calculate its direction cosines
\\
\solution \\
\begin{table}[h!]    
		\centering
		\input{tables/table.tex}
		\caption{}
\end{table} 
\begin{align}
     \overrightarrow{a} &= \myvec{1 \\ 1\\ -1}
\end{align}
The direction ratios are 1,1,-1\\
\begin{align}
     \norm{a} &= \sqrt{\brak{a^Ta}}\\
     \norm{a} &= \sqrt{3}\\
     a_{\text{normalized}} &= {\frac{1}{\sqrt3}}\myvec{1\\1\\-1}
\end{align}
The direction cosines are $\frac{1}{\sqrt3}, \frac{1}{\sqrt3}, \frac{-1}{\sqrt{3}}$
