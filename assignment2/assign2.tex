\textbf{Question}:\\
The midpoint of line segment joining $\vec{A}\brak{2a,4}$ and$\vec{B}\brak{-2,3b}$ is $\vec{C}\brak{1,2a+1}$. Find the values of a and b.\\
\textbf{Solution}:\\
\begin{table}[h!]
   \centering
   \input{tables/table.tex}
   \caption{variables used}
   \label{tabQuestion-1-1.5-17}
\end{table}   
\begin{align}
    \vec{C}=\frac{\vec{A}+\vec{B}}{2}
\end{align}
Now substituting the values of $\vec{A}$, $\vec{B}$ and $\vec{C}$ 
\begin{align}
     \myvec {1\\2a+1}=\frac{\myvec{2a\\4}+\myvec{-2\\3b}}{2}\\
     \myvec{1\\2a+1}=\myvec{a-1\\4+\frac{3b}{2}}
\end{align}
\begin{align}
     a-1=1\\
    \implies a=2
\end{align} 
\begin{align}
     2a+1=a+\frac{3b}{2}\\
    \implies b=2\\
\end{align}
